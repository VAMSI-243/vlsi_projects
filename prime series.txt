module prime(n,y);
input [8:0] n; //range of 300
output reg y;
integer i;
always @ (*) begin
y=1'b1;
if (n<2) begin
y=1'b0;
end
else  begin
for (i=2;i<n;i=i+1) begin
if(n%i==0)
   y=1'b0;
   end
   end
   end
   endmodule

module tb;
reg [8:0]n;
wire y;
prime dut(.*);
initial begin
//$monitor($time,"n=%0d,y=%0d",n,y);
for(n=0;n<300;n=n+1) begin
#1;
if (y)
$display($time,"n=%0d,y=%0d",n,y);
end
$finish; 
end

endmodule
